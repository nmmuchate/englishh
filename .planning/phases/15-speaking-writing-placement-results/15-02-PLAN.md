This is an onCall function that takes all stage results and computes the final CEFR placement:

```javascript
// ── Phase 15: Final placement calculation ────────────────────────────────

// CEFR level numeric mapping for averaging
const CEFR_VALUES = { 'A1': 1, 'A2': 2, 'B1': 3, 'B2': 4, 'C1': 5, 'C2': 6 }
const VALUE_TO_CEFR = { 1: 'A1', 2: 'A2', 3: 'B1', 4: 'B2', 5: 'C1', 6: 'C2' }
const CEFR_NAMES = { 'A1': 'Beginner', 'A2': 'Elementary', 'B1': 'Intermediate', 'B2': 'Upper Intermediate', 'C1': 'Advanced', 'C2': 'Proficiency' }

exports.calculatePlacement = onCall({ region: 'africa-south1' }, async (request) => {
  if (!request.auth) {
    throw new HttpsError('unauthenticated', 'Authentication required')
  }

  const { stageResults = {} } = request.data
  const uid = request.auth.uid

  // Extract per-skill levels — default to B1 for skipped/missing stages
  const vocabLevel = stageResults.vocabulary?.level || 'B1'
  const listeningLevel = stageResults.listening?.level || 'B1'
  const grammarLevel = stageResults.grammar?.level || 'B1'
  const speakingLevel = stageResults.speaking?.level || 'B1'
  const writingLevel = stageResults.writing?.level || 'B1'
  // Reading level from vocabulary stage (vocabulary+reading are combined)
  const readingLevel = stageResults.vocabulary?.level || 'B1'

  const skillBreakdown = {
    vocabulary: vocabLevel,
    reading: readingLevel,
    listening: listeningLevel,
    grammar: grammarLevel,
    speaking: speakingLevel,
    writing: writingLevel
  }

  // Calculate overall by averaging numeric values and rounding
  const values = Object.values(skillBreakdown).map(l => CEFR_VALUES[l] || 3)
  const avg = values.reduce((a, b) => a + b, 0) / values.length
  const rounded = Math.round(avg)
  const overallLevel = VALUE_TO_CEFR[Math.min(Math.max(rounded, 1), 6)] || 'B1'

  // Determine strengths (highest skills) and weaknesses (lowest skills)
  const sorted = Object.entries(skillBreakdown)
    .map(([skill, level]) => ({ skill, value: CEFR_VALUES[level] || 3 }))
    .sort((a, b) => b.value - a.value)
  const strengths = sorted.slice(0, 2).map(s => s.skill)
  const weaknesses = sorted.slice(-2).map(s => s.skill)

  const result = {
    overallLevel,
    overallName: CEFR_NAMES[overallLevel] || 'Intermediate',
    skillBreakdown,
    strengths,
    weaknesses,
    confidence: Object.values(stageResults).filter(r => !r.skipped).length >= 3 ? 'high' : 'low'
  }

  // Persist final result to user document + placementTests
  try {
    const now = Timestamp.now()
    await Promise.all([
      admin.firestore().doc(`users/${uid}`).update({
        placement: result,
        currentLevel: overallLevel,
        onboardingCompleted: true,
        updatedAt: now
      }),
      admin.firestore().doc(`placementTests/${uid}`).set({
        finalResult: result,
        completedAt: now
      }, { merge: true })
    ])
  } catch (err) {
    logger.error('calculatePlacement: Firestore write failed', { uid, error: err.message })
    // Still return result even if persistence fails — client can retry save
  }

  logger.info('calculatePlacement complete', { uid, overallLevel })
  return result
})
```

```javascript
const error = ref(null)

// cefrToValue maps for progress bars
function cefrToValue(level) {
  const map = { A1: 0.1, A2: 0.2, B1: 0.4, B2: 0.6, C1: 0.8, C2: 1.0 }
  return map[level] || 0.4
}

// Skill order for display
const skills = computed(() => {
  if (!result.value?.skillBreakdown) return []
  const order = ['vocabulary', 'reading', 'listening', 'grammar', 'speaking', 'writing']
  return order.map(key => ({
    name: key.charAt(0).toUpperCase() + key.slice(1),
    level: result.value.skillBreakdown[key] || 'B1'
  }))
})

const strengthsText = computed(() => {
  if (!result.value?.strengths) return ''
  return result.value.strengths.map(s => s.charAt(0).toUpperCase() + s.slice(1)).join(', ')
})

const weaknessesText = computed(() => {
  if (!result.value?.weaknesses) return ''
  return result.value.weaknesses.map(s => s.charAt(0).toUpperCase() + s.slice(1)).join(', ')
})

const levelName = computed(() => {
  const names = { A1: 'Beginner', A2: 'Elementary', B1: 'Intermediate', B2: 'Upper Intermediate', C1: 'Advanced', C2: 'Proficiency' }
  return names[result.value?.overallLevel] || 'Intermediate'
})

onMounted(() => {
```

```javascript
async function handleSpeakingComplete(result) {
  isSaving.value = true
  saveError.value = null
  try {
    placementStore.setStageResult('speaking', { score: result.score || 0, level: result.level || placementStore.adaptiveLevel, skipped: false, transcript: result.transcript })
    writingPrompt.value = result.writingPrompt || ''
    step.value = 'writing'
  } catch (err) {
    console.error('handleSpeakingComplete failed:', err)
    saveError.value = err.message || 'Failed to save speaking results. Please try again.'
  } finally {
    isSaving.value = false
  }
}

function handleSpeakingSkip() {
  placementStore.setStageResult('speaking', { score: 0, level: 'B1', skipped: true, transcript: [] })
  writingPrompt.value = ''
  step.value = 'writing'
}

async function handleWritingComplete(result) {
  isSaving.value = true
  saveError.value = null
  try {
    // Step 1: Call evaluateSpeakingTest with speaking transcript + writing response
    const speakingResult = placementStore.stageResults.speaking || {}
    const evaluateFn = httpsCallable(functions, 'evaluateSpeakingTest')
    const evalResponse = await evaluateFn({
      speakingTranscript: speakingResult.transcript || [],
      writingResponse: result.response || '',
      currentLevel: placementStore.adaptiveLevel
    })
    const evalData = evalResponse.data

    // Update speaking + writing stage results with AI evaluation scores
    if (evalData.speaking) {
      placementStore.setStageResult('speaking', {
        ...speakingResult,
        score: evalData.speaking.score,
        level: evalData.speaking.level,
        evaluation: evalData.speaking
      })
    }
    placementStore.setStageResult('writing', {
      response: result.response,
      score: evalData.writing?.score || 0,
      level: evalData.writing?.level || placementStore.adaptiveLevel,
      skipped: false,
      evaluation: evalData.writing
    })

    // Step 2: Call calculatePlacement with all stage results
    const calcFn = httpsCallable(functions, 'calculatePlacement')
    const calcResponse = await calcFn({ stageResults: placementStore.stageResults })
    const placementResult = calcResponse.data

    // Step 3: Store final result and navigate to results page
    await placementStore.completeTest(placementResult)
    router.push({ name: 'placement-result' })
  } catch (err) {
    console.error('handleWritingComplete failed:', err)
    saveError.value = err.message || 'Failed to calculate placement. Please try again.'
  } finally {
    isSaving.value = false
  }
}

function handleWritingSkip() {
  // Skip writing — still need to evaluate speaking (if not skipped) and calculate placement
  placementStore.setStageResult('writing', { score: 0, level: 'B1', skipped: true, response: '' })
  // Trigger placement calculation
  handlePlacementCalculation()
}

async function handlePlacementCalculation() {
  isSaving.value = true
  saveError.value = null
  try {
    // If speaking was not skipped, evaluate it even though writing was skipped
    const speakingResult = placementStore.stageResults.speaking || {}
    if (!speakingResult.skipped && speakingResult.transcript?.length) {
      const evaluateFn = httpsCallable(functions, 'evaluateSpeakingTest')
      const evalResponse = await evaluateFn({
        speakingTranscript: speakingResult.transcript || [],
        writingResponse: '',
        currentLevel: placementStore.adaptiveLevel
      })
      const evalData = evalResponse.data
      if (evalData.speaking) {
        placementStore.setStageResult('speaking', {
          ...speakingResult,
          score: evalData.speaking.score,
          level: evalData.speaking.level,
          evaluation: evalData.speaking
        })
      }
    }

    const calcFn = httpsCallable(functions, 'calculatePlacement')
    const calcResponse = await calcFn({ stageResults: placementStore.stageResults })
    const placementResult = calcResponse.data

    await placementStore.completeTest(placementResult)
    router.push({ name: 'placement-result' })
  } catch (err) {
    console.error('handlePlacementCalculation failed:', err)
    saveError.value = err.message || 'Failed to calculate placement. Please try again.'
  } finally {
    isSaving.value = false
  }
}
```
