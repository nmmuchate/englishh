**2. Add evaluateSpeakingTest export** — new onCall function after generateTestQuestions:

```javascript
exports.evaluateSpeakingTest = onCall({ region: 'africa-south1', secrets: [OPENAI_API_KEY] }, async (request) => {
  if (!request.auth) {
    throw new HttpsError('unauthenticated', 'Authentication required')
  }

  const { speakingTranscript = [], writingResponse = '', currentLevel = 'B1' } = request.data

  if (!speakingTranscript.length && !writingResponse) {
    throw new HttpsError('invalid-argument', 'At least speakingTranscript or writingResponse required')
  }

  const openai = new OpenAI({ apiKey: OPENAI_API_KEY.value() })

  const transcriptText = speakingTranscript
    .map(t => `${t.speaker}: ${t.text}`)
    .join('\n')

  const response = await openai.chat.completions.create({
    model: 'gpt-4o-mini',
    temperature: 0.3,
    max_tokens: 800,
    response_format: { type: 'json_object' },
    messages: [
      {
        role: 'system',
        content: 'You are an expert English language assessor. Evaluate the student responses and assign CEFR levels. Always respond with valid JSON only.'
      },
      {
        role: 'user',
        content: `Evaluate this student's speaking and writing for CEFR placement.
Current estimated level: ${currentLevel}

Speaking transcript:
${transcriptText || '(skipped)'}

Writing response:
${writingResponse || '(skipped)'}

Return JSON:
{
  "speaking": {
    "level": "B1",
    "score": 65,
    "fluency": 70,
    "pronunciation": 60,
    "coherence": 65,
    "feedback": "Brief 1-sentence feedback"
  },
  "writing": {
    "level": "B1",
    "score": 60,
    "grammar": 65,
    "vocabulary": 55,
    "coherence": 60,
    "feedback": "Brief 1-sentence feedback"
  }
}

Rules:
- score is 0-100 overall for that skill
- level is one of A1, A2, B1, B2, C1, C2
- If speaking transcript is "(skipped)", set speaking.level to "${currentLevel}" and speaking.score to 50
- If writing response is "(skipped)", set writing.level to "${currentLevel}" and writing.score to 50
- Sub-scores (fluency, pronunciation, coherence, grammar, vocabulary) are 0-100`
      }
    ]
  })

  let parsed
  try {
    parsed = JSON.parse(response.choices[0].message.content)
  } catch {
    throw new HttpsError('internal', 'Evaluation failed — please retry')
  }

  logger.info('evaluateSpeakingTest complete', { uid: request.auth.uid, speakingLevel: parsed.speaking?.level, writingLevel: parsed.writing?.level })
  return parsed
})
```
